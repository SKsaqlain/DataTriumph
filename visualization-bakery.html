itemsets=pd.DataFrame(columns=['Item'])
item=processed['Items']
i=0;
for x in item:
    if(len(x.split('|'))>1):
        itemsets.loc[i]=x
        i+=1
itemset_counts=itemsets['Item'].value_counts()

top=itemset_counts[:10]
